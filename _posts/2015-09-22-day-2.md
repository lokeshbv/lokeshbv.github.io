## Ruby Control Stuctures

### Each
{% highlight ruby linenos %}

def scoring(array)
    array.each do |user|
        user.update_score
    end
end

{% endhighlight %}

### Unless
{% highlight ruby linenos %}

def scoring(array)
    array.each do |user|
        unless user.is_admin?
            user.update_score
        end
    end
end

{% endhighlight %}

### Case (Bonus Question)

def identify_class(obj)
    case obj
    when Hacker
        puts "It's a Hacker!"
    when Submission
        puts "It's a Submission!"
    when TestCase
        puts "It's a TestCase!"
    when Contest
        puts "It's a Contest!"
    else
        puts "It's an unknown model"
    end
end
